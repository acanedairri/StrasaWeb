SELECT DISTINCT		
		t1.datavalue as Location, 
		t2.datavalue as Country,
		t3.datavalue as Province,
		t4.datavalue as Region,
		t5.datavalue as Altitude,
		t6.datavalue as Latitude,
		t7.datavalue as WeatherStation

FROM studyrawdata t1 
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Country') t2 ON t1.datarow = t2.datarow
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Province') t3 ON t1.datarow = t3.datarow
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Region') t4 ON t1.datarow = t4.datarow
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Altitude') t5 ON t1.datarow = t5.datarow
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Latitude') t6 ON t1.datarow = t6.datarow
		LEFT JOIN (SELECT datavalue,datarow FROM studyrawdata WHERE datacolumn='Weather Station') t7 ON t1.datarow = t7.datarow
		
WHERE t1.studyid = 1 AND t1.datacolumn = 'Location'
